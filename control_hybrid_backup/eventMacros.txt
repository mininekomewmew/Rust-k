automacro testchat_perfect {
exclusive 1

call {
pause 5
        # 8C 00 = ID
        # 68 69 00 00 00 00 00 00 00 00 00 00 = "hi" padded to 12 bytes
        # Total = 14 bytes EXACTLY.

}

# --- Morocc Buff NPC After Respawn ---

automacro getBuffAfterDeath {
    # This triggers whenever you load into Morocc (e.g., after respawning)
    MapLoaded morocc
    exclusive 1
    call {
        # Wait a bit to ensure everything is loaded
        pause 3
        
        # --- Change the coordinates below to your NPC's location ---
        # Example: do move 160 100
        do move 160 100
        
        # --- Change the coordinates and sequence below ---
        # c = continue, r0 = first response, r1 = second response, etc.
        # Example: talk to NPC at 160 100, then click Next (c) and Choose first option (r0)
        do talknpc 160 100 c r0
    }
}
